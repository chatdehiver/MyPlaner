CREATE SEQUENCE message_seq
		start with 1
	        increment by 1
		nocache;

CREATE TABLE message (
	message_no NUMBER PRIMARY KEY,
	sender_nickname VARCHAR2(30),
	destination_nickname VARCHAR2(30),
	message_send_date DATE NOT NULL,
	message_content VARCHAR2(1000) NOT NULL,
	message_title VARCHAR2(100) NOT NULL,
	message_check VARCHAR2(1) DEFAULT 0,
	CONSTRAINT fk_message1 FOREIGN KEY (sender_nickname) REFERENCES member(nickname) ON DELETE CASCADE,
	CONSTRAINT fk_message2 FOREIGN KEY (destination_nickname) REFERENCES member(nickname) ON DELETE CASCADE
);
