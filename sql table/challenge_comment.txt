CREATE SEQUENCE challenge_comment_seq
		start with 1
	        increment by 1
		nocache;

CREATE TABLE challenge_comment (
	challenge_comment_no NUMBER PRIMARY KEY,
	challenge_no NUMBER,
	challenge_reply_comment NUMBER NULL,
	challenge_comment_nickname VARCHAR2(30) NOT NULL,
	challenge_comment VARCHAR2(1000) NOT NULL,
	challenge_comment_write_date DATE NOT NULL,
	CONSTRAINT FK_ccomment1 FOREIGN KEY (challenge_no) REFERENCES challenge(challenge_no) ON DELETE CASCADE,
	CONSTRAINT FK_ccomment2 FOREIGN KEY (challenge_reply_comment) REFERENCES challenge_comment(challenge_comment_no) ON DELETE CASCADE,
	CONSTRAINT FK_ccomment3 FOREIGN KEY (challenge_comment_nickname) REFERENCES member(nickname) ON DELETE CASCADE
);
