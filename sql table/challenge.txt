CREATE TABLE challenge (
	challenge_no NUMBER PRIMARY KEY,
	challenge_title VARCHAR2(50) NOT NULL,
	challenge_start_date DATE NOT NULL,
	challenge_end_date DATE,
	nickname VARCHAR2(30),
	challengeCategory VARCHAR2(30) NOT NULL,
	challengeLikes NUMBER DEFAULT 0,
	challengeSharing NUMBER DEFAULT 0,
	CONSTRAINT FK_challenge FOREIGN KEY (nickname) REFERENCES member(nickname) ON DELETE CASCADE
);
